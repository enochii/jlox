// we implement for-loop as a sugar,

for(var i = 0; i < 3; i=i+1) {
    println("before continue");
    continue;
}

// and the corresponding while loop is like:
{
    var i = 0;
    while(i < 3) {
        println("before continue");
        continue;
        // in this case, `i=i+1;` statement will never be executed!
        // TODO: fix this
        i = i+1;
    }
}
